**Test 10: Infinite loop guard (interface test)**
Define the interface contract that Phase 9's stop hook must implement:
```javascript
// Interface: stop hook must check a flag to prevent infinite recursion
const stopHookInterface = {
  checkFlag: (sessionId) => {
    const flagPath = path.join(os.tmpdir(), `dynamo-stop-active-${sessionId}`);
    return fs.existsSync(flagPath);
  },
  setFlag: (sessionId) => {
    const flagPath = path.join(os.tmpdir(), `dynamo-stop-active-${sessionId}`);
    fs.writeFileSync(flagPath, '1');
  },
  clearFlag: (sessionId) => {
    const flagPath = path.join(os.tmpdir(), `dynamo-stop-active-${sessionId}`);
    try { fs.unlinkSync(flagPath); } catch {}
  }
};
// Test the interface
const testId = 'regression-test-' + Date.now();
assert.strictEqual(stopHookInterface.checkFlag(testId), false, 'Flag should not exist initially');
stopHookInterface.setFlag(testId);
assert.strictEqual(stopHookInterface.checkFlag(testId), true, 'Flag should exist after set');
stopHookInterface.clearFlag(testId);
assert.strictEqual(stopHookInterface.checkFlag(testId), false, 'Flag should be cleared');
```

       ```javascript
       'use strict';

       const PARSERS = Object.freeze({
         v1: Object.freeze({
           detect(parsed) {
             return Array.isArray(parsed.chat_messages)
               && parsed.chat_messages.length > 0
               && typeof parsed.chat_messages[0].sender === 'string';
           },

           parseConversation(conversation) {
             return {
               id: conversation.uuid || 'unknown',
               title: conversation.name || 'Untitled',
               created: conversation.created_at || null,
               updated: conversation.updated_at || null,
               turns: (conversation.chat_messages || []).map(function (msg, idx) {
                 return {
                   index: idx,
                   sender: msg.sender,
                   text: extractTextContent(msg.content),
                   timestamp: msg.created_at || null,
                 };
               }),
             };
           },
         }),
       });
       ```

       Functions:
       a. extractTextContent(contentArray):
          - If not array, return ''
          - Filter items where item.type === 'text'
          - Map to item.text, join with '\n'

       b. detectVersion(parsed):
          - Iterate PARSERS, call detect, return first match or null

       c. parseExportFile(jsonString):
          - JSON.parse the input
          - Handle single conversation or array of conversations
          - Detect version via detectVersion
          - If no version matches, return err('UNKNOWN_FORMAT', ...)
          - Return ok({ conversations: [...], version: 'v1', count: N })

       d. getConversationAge(createdAt):

    'use strict';
    const { ok, err } = require('../../../../lib/result.cjs');
    const { BACKFILL_DEFAULTS, FORMATION_DEFAULTS } = require('../../lib/constants.cjs');
    const { createBackfillParser } = require('./backfill-parser.cjs');

    Export createBackfillPipeline(options) factory.
    Options: { formationPipeline, selfModel, switchboard, lathe, config }

    Functions:

    1. dryRun(exportJson):
       - Parse via backfillParser.parseExportFile(exportJson)
       - If parse fails, return err
       - Count conversations, total turns, estimate fragments (turns * attention_pass_rate_estimate)
       - Return ok({ conversations: N, total_turns: N, user_turns: N, estimated_fragments: estimate, oldest: date, newest: date })
       - NO writes, NO formation calls

    2. processConversation(conversation, options):
       - conversation: parsed conversation object from backfillParser
       - options: { dryRun: false, maxFragments: BACKFILL_DEFAULTS.max_fragments_per_conversation }
       - Extract user turns only (sender === 'human')
       - For each user turn:
         a. Get the assistant response that follows (if any) for context
         b. Compute temporal values per Pitfall 3:
            - temporal.absolute = turn.timestamp (original message time)
            - temporal.session_relative = turn.index / totalTurns (normalized 0-1)
            - temporal.sequence = turn.index
         c. Compose the backfill-specific formation prompt using BACKFILL_TEMPLATES:
            ```javascript
            // Get conversation age for framing context
            const conversationAge = _parser.getConversationAge(conversation.created);
            const selfModelSnapshot = _selfModel ? _selfModel.getAspect('identity-core') : null;

            // Compose the full backfill prompt from BACKFILL_TEMPLATES
            const backfillSystemPrompt = BACKFILL_TEMPLATES.backfill_formation.system;
            const backfillUserPrompt = BACKFILL_TEMPLATES.backfill_formation.user(
              turn, selfModelSnapshot, conversationAge
            );
            ```
         d. Build stimulus package with the pre-composed backfill prompt:
            ```javascript
            const stimulus = {
              user_prompt: turn.text,
              assistant_context: nextAssistantTurn ? nextAssistantTurn.text : '',
              turn_number: turn.index,
              session_summary: `Backfill: "${conversation.title}" (${conversationAge})`,
              origin: 'backfill',
              // Backfill-specific: pre-composed prompt override for formation
              backfill_prompt: {
                system: backfillSystemPrompt,
                user: backfillUserPrompt,
              },
              backfill_temporal: {
                absolute: turn.timestamp,
                session_relative: turn.index / Math.max(1, conversation.turns.length - 1),
                sequence: turn.index,
              },
            };
            ```
            The backfill_prompt field carries the pre-composed BACKFILL_TEMPLATES content
            into the formation pipeline. The formation pipeline's prepareStimulus checks for
            stimulus.backfill_prompt and uses it as the formation prompt instead of the default
            FORMATION_TEMPLATES. This approach keeps backfill-specific framing in backfill-pipeline
            while reusing the existing formation infrastructure for attention gating and output processing.
         e. Pass through formationPipeline.prepareStimulus to check attention gate
         f. If attention gate passes, invoke formation (via processFormationOutput)
         g. Post-formation: patch the fragment frontmatter to set:
            - origin: 'backfill' (per D-14)
            - source_session: `backfill-${conversation.id}` (per Pitfall 3)
            - temporal.absolute: original timestamp (override if formation sets current time)
         h. Track fragment count; stop if exceeds maxFragments cap
       - Emit progress: switchboard.emit('reverie:backfill:progress', { conversation_id, turn, total_turns, fragments_formed })
       - Return { turns_processed, fragments_formed, skipped_by_attention }

    3. runBatch(exportJson, options):
       - options: { dryRun: false, batchSize: BACKFILL_DEFAULTS.default_batch_size, limit: null, conversationId: null }
       - Parse export file
       - If conversationId specified, filter to single conversation
       - If limit specified, take first N conversations
       - Process in batches of batchSize
       - Aggregate stats across all conversations
       - Emit batch progress: switchboard.emit('reverie:backfill:batch-progress', { current, total, stats })
       - Return ok({ conversations_processed, turns_processed, fragments_formed, skipped_by_attention, errors: [] })

    4. BACKFILL_TEMPLATES integration path: backfill-pipeline.cjs requires BACKFILL_TEMPLATES from prompt-templates.cjs. For each conversation turn, it calls BACKFILL_TEMPLATES.backfill_formation.system (string) and BACKFILL_TEMPLATES.backfill_formation.user(turn, selfModel, age) to compose the complete backfill prompt. This composed prompt is passed as stimulus.backfill_prompt to formationPipeline.prepareStimulus, which uses it as the formation prompt override (instead of the default FORMATION_TEMPLATES prompts). The formation pipeline handles attention gating and output processing as normal.

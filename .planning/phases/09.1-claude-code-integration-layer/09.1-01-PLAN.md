```javascript
async function writeFileAtomic(filePath, content) {
  const tmpPath = filePath + '.tmp';
  try {
    fs.mkdirSync(path.dirname(filePath), { recursive: true });
    await Bun.write(tmpPath, content);
    fs.renameSync(tmpPath, filePath);
    return ok(undefined);
  } catch (e) {
    try { fs.unlinkSync(tmpPath); } catch (_) {}
    return err('WRITE_FAILED', `Atomic write failed: ${e.message}`, { path: filePath });
  }
}
```

Transcript JSONL structure (from research):
```
Each line is a JSON object:
- User turns: { "role": "user", "content": [...] }
- Assistant turns: { "role": "assistant", "content": [...] }
- Content arrays contain objects with "type": "text" | "tool_use" | "tool_result"
- Tool use: { "type": "tool_use", "id": "toolu_xxx", "name": "Read", "input": {...} }
- Tool result: { "type": "tool_result", "tool_use_id": "toolu_xxx", "content": "..." }
```

Hook payload (provides transcript_path):
```json
{
  "session_id": "claude-session-123",
  "transcript_path": "/Users/user/.claude/projects/proj-id/session-id.jsonl",
  "tool_name": "Read",
  "tool_input": { "file_path": "/src/index.ts" },
  "tool_response": "file contents...",
  "tool_use_id": "toolu_abc123"
}
```

Build the facade generator (FWK-02) and hook definitions with wiring registry (FWK-05). Facades wrap resolved services/providers with hook points for plugin extension. Hook definitions canonicalize all 8 Claude Code hook types and provide a declarative wiring mechanism.

    The facade generator creates a NEW object that delegates method calls to the underlying frozen contract. It does NOT modify the contract. Implementation per D-06 and D-08:

    ```
    function createFacade(name, contract, metadata = {}) {
      const _hooks = {};      // { methodName: { before: [], after: [], around: [] } }
      const _overrides = {};   // { methodName: replacementFn }

      function hook(method, position, handler) {
        // position must be 'before', 'after', or 'around'
        if (!_hooks[method]) _hooks[method] = { before: [], after: [], around: [] };
        if (!_hooks[method][position]) {
          return err('FACADE_INVALID_POSITION', `Invalid hook position "${position}"`);
        }
        _hooks[method][position].push(handler);
        return ok(undefined);
      }

      function override(method, newImpl) {
        if (typeof contract[method] !== 'function') {
          return err('FACADE_INVALID_METHOD', `Method "${method}" not found on "${name}"`);
        }
        _overrides[method] = newImpl;
        return ok(undefined);
      }

      // Build facade by iterating contract keys
      const facade = {};
      facade.meta = Object.freeze({ name, ...metadata });

      for (const key of Object.keys(contract)) {
        if (typeof contract[key] === 'function') {
          facade[key] = function(...args) {
            const hooks = _hooks[key] || { before: [], after: [], around: [] };
            const impl = _overrides[key] || contract[key];

            // Around hooks: wrap the call chain
            if (hooks.around.length > 0) {
              let index = 0;
              function next(currentArgs) {
                if (index < hooks.around.length) {
                  const aroundHandler = hooks.around[index++];
                  return aroundHandler(next, currentArgs);
                }
                // After around chain, run before/impl/after as normal
                return executeCore(impl, currentArgs, hooks);
              }
              return next(args);
            }

            return executeCore(impl, args, hooks);
          };
        } else {
          facade[key] = contract[key];  // Non-function properties copied as-is
        }
      }

      function executeCore(impl, args, hooks) {
        // Before hooks: can modify args (return array) or halt (return Err)
        for (const h of hooks.before) {
          const result = h(args, name, key_unused_here);
          // Use the method name from closure
          if (result && result.ok === false) return result;  // Halt
          if (Array.isArray(result)) args = result;           // Modify args
        }

        // Execute implementation
        let result = impl(...args);

        // After hooks: can modify return value
        for (const h of hooks.after) {
          const modified = h(result, args, name, key_unused_here);
          if (modified !== undefined) result = modified;
        }

        return result;
      }

      // Note: executeCore uses method-specific key from the closure in the facade[key] assignment.
      // Refactor: pass the method key into executeCore. The actual implementation should capture `key` in the for-loop closure properly (use `const` in the for-of loop which creates a new binding per iteration).

      facade.hook = hook;
      facade.override = override;

      return Object.freeze(facade);
    }
    ```

    CRITICAL implementation detail: The for...of loop with `const key` creates a new binding per iteration, so each facade method closure captures its own `key`. Use this pattern:
    ```javascript
    for (const key of Object.keys(contract)) {
      if (typeof contract[key] === 'function') {
        facade[key] = createMethodProxy(key, contract[key]);
      }
    }
    ```
    Where `createMethodProxy(methodName, originalFn)` returns a function that:
    1. Checks around hooks first (chain pattern with `next`)
    2. Runs before hooks (halt on Err, replace args on Array return)
    3. Calls `_overrides[methodName] || originalFn` with processed args
    4. Runs after hooks (replace result if handler returns non-undefined)
    5. Returns final result

    Export: `module.exports = { createFacade }`

<objective>
Build the Relay install/update/sync orchestration service: orchestrates platform self-install, self-update, and sync operations using Forge for git operations and Lathe for file operations. Every operation follows backup-before-modify with git-tag-based rollback on failure per D-10. Relay also manages config migration between versions per D-09 and plugins/modules as git submodules via Forge per D-08.

Purpose: Relay is the operations orchestrator. It ensures that install, update, and sync operations are safe (rollback on failure) and that the platform can self-manage its lifecycle. Forge provides the git primitives; Relay composes them into safe, atomic operations.

Output: Relay service factory, full test suite.
</objective>

    This is the core backup-modify-verify-rollback pattern per D-10:
    ```javascript
    async function _withBackup(operationName, fn) {
      const tagName = `relay-backup-${operationName}-${Date.now()}`;

      // 1. Create backup tag
      const tagResult = _forge.tag(tagName);
      if (!tagResult.ok) {
        return err('BACKUP_FAILED', `Failed to create backup tag: ${tagResult.error.message}`);
      }

      try {
        // 2. Execute operation
        const result = await fn();

        if (!result.ok) {
          // 3a. Rollback on failure
          _forge.resetTo(tagName);
          _forge.deleteTag(tagName);
          return result;
        }

        // 3b. Clean up tag on success
        _forge.deleteTag(tagName);
        return result;
      } catch (e) {
        // 3c. Rollback on exception
        _forge.resetTo(tagName);
        _forge.deleteTag(tagName);
        return err('OPERATION_FAILED', `${operationName} failed: ${e.message}`);
      }
    }
    ```

    ```javascript
      return _withBackup('install', async () => {
        // Sync files from src to dest
        const syncResult = await _forge.sync(srcDir, destDir);
        if (!syncResult.ok) return syncResult;

        const stageResult = _forge.stageAll();
        if (!stageResult.ok) return stageResult;

        const commitResult = _forge.commit(`relay: install from ${srcDir}`);
        if (!commitResult.ok) return commitResult;

        // Emit event
        if (_switchboard) {
          _switchboard.emit('relay:installed', { srcDir, destDir, hash: commitResult.value.hash });
        }

        return ok({ hash: commitResult.value.hash });
      });
    }
    ```

    **Config migration (per D-09):**
    ```javascript
    async function migrateConfig(currentConfig, targetVersion) {
      // Merge new defaults into existing config, preserving user values
      // Default config structure includes all known keys with default values
      const defaults = _getDefaultConfig(targetVersion);

      // Deep merge: current config values take precedence over defaults
      const merged = {};
      // First apply defaults
      for (const key of Object.keys(defaults)) {
        merged[key] = defaults[key];
      }
      // Then overlay current config (preserves user values)
      for (const key of Object.keys(currentConfig || {})) {
        if (key === '_version') continue; // version will be stamped
        if (typeof currentConfig[key] === 'object' && currentConfig[key] !== null && !Array.isArray(currentConfig[key])) {
          merged[key] = { ...(defaults[key] || {}), ...currentConfig[key] };
        } else {
          merged[key] = currentConfig[key];
        }
      }
      // Stamp version
      merged._version = targetVersion;

      return ok(merged);
    }

    function _getDefaultConfig(version) {
      // Return default config shape for the given version
      // This is extensible -- each version can define its own defaults
      return {
        _version: version,
        platform: { name: 'dynamo' },
        services: {},
        providers: {},
        modules: {},
        plugins: {},
      };
    }
    ```
